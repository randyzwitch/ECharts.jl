using ECharts
data = [
        Dict(
            "name" => "A",            
            "value" => 30,
            "children" => [Dict(
                "name" => "A1",
                "value" => 14
            ), Dict(
                "name" => "A2",
                "value" => 16
            )]
        ),
        Dict(
            "name" => "B",
            "value" => 34,
            "children" => [Dict(
                "name" => "B1",
                "value" => 20,
                "children" => [Dict(
                    "name" => "B2",
                    "value" => 15
                )]
            )]
        )
    ]

sb = sunburst(data)
